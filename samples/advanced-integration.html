Integration Code

<script src="../dist/chatbot-widget.min.js"></script>
<script>
  // Initialize with custom options
  ChatbotWidget.init({
    apiUrl: 'http://localhost:5000',
    botName: 'Custom Assistant',
    primaryColor: '#4CAF50',
    welcomeMessage: 'Hello! I\'m your custom assistant. How can I help you today?',
    placeholderText: 'Ask me anything...'
  });
  
  // Programmatically control the widget
  document.getElementById('openChat').addEventListener('click', () => {
    ChatbotWidget.open();
  });
  
  document.getElementById('closeChat').addEventListener('click', () => {
    ChatbotWidget.close();
  });
  
  document.getElementById('clearChat').addEventListener('click', () => {
    ChatbotWidget.clearConversation();
  });
</script>
